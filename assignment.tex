\subsection{Surface Integral}
\begin{eqnarray}
\iint_Sf(x,y)dxdy\\
\oiint_S f(x,y)dxdy\\
\int_{x_1}^{x_2}\int_{y_1}^{y_2}f(x,y)dxdy
\end{eqnarray}

\subsection{Volume Integral}
\begin{eqnarray}
\iiint_Vf(x,y,z)dxdydz\\
\int_{x_1}^{x_2}\int_{y_1}^{y_2}\int_{z_1}^{z_2}f(x,y,z)dxdydz
\end{eqnarray}
